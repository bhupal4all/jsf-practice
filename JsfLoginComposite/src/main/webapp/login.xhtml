	<composite:attribute name="passwordValue" />
	<composite:attribute name="loginLabel" default="d-Submit"
		displayName="Submit" />

	<!-- Error Message -->
	<composite:attribute name="showMessgeDetails" default="false"

	<composite:editableValueHolder name="password"

		targets="loginForm:username loginForm:password" />

		method-signature="java.lang.String action()" />
</composite:interface>
