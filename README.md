JSF Practice

# Create Simple Web App
mvn archetype:create -DgroupId=com.ranga.jsf -DartifactId=JsfPractice -DarchetypeArtifactId=maven-archetype-webapp

# Eclipse Project creation
mvn eclipse:eclipse -Dwtpversion=2.0
